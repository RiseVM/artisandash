# main.py
import os
import sqlite3
from datetime import datetime, date
from flask import Flask, request, redirect, url_for, render_template_string, flash

app = Flask(__name__)
app.secret_key = os.environ.get("FLASK_SECRET_KEY", "dev-secret")  # set in Replit env vars for production

DB_PATH = "samples.db"

# Email / notification configuration via environment variables
SMTP_HOST = os.environ.get("SMTP_HOST", "")
SMTP_PORT = int(os.environ.get("SMTP_PORT", "587"))
SMTP_USER = os.environ.get("SMTP_USER", "")
SMTP_PASS = os.environ.get("SMTP_PASS", "")
FROM_EMAIL = os.environ.get("FROM_EMAIL", "")

TEAM_EMAIL = os.environ.get("TEAM_EMAIL", "")  # Artisan Tile team notifications
TEAM_NAME = os.environ.get("TEAM_NAME", "Artisan Tile Team")


def get_db():
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn


def init_db():
    conn = get_db()
    c = conn.cursor()
    c.execute(
        """
        CREATE TABLE IF NOT EXISTS samples (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            customer_name TEXT NOT NULL,
            customer_email TEXT NOT NULL,
            customer_phone TEXT,
            sample_name TEXT NOT NULL,
            checkout_date TEXT NOT NULL,
            due_date TEXT NOT NULL,
            status TEXT NOT NULL DEFAULT 'checked_out',
            notification_sent INTEGER NOT NULL DEFAULT 0,
            team_notification_sent INTEGER NOT NULL DEFAULT 0,
            notes TEXT,
            auth_notes TEXT,
            stripe_customer_id TEXT,
            created_at TEXT NOT NULL,
            updated_at TEXT NOT NULL
        )
        """
    )
    conn.commit()
    conn.close()


def send_email(to_email, subject, body):
    """Basic SMTP email sender. Config with env vars.
    If not configured, it will just print to console.
    """
    if not (SMTP_HOST and SMTP_USER and SMTP_PASS and FROM_EMAIL):
        print("=== EMAIL (simulation) ===")
        print("To:", to_email)
        print("Subject:", subject)
        print("Body:\n", body)
        print("=========================")
        return

    import smtplib
    from email.mime.text import MIMEText

    msg = MIMEText(body)
    msg["From"] = FROM_EMAIL
    msg["To"] = to_email
    msg["Subject"] = subject

    try:
        with smtplib.SMTP(SMTP_HOST, SMTP_PORT) as server:
            server.starttls()
            server.login(SMTP_USER, SMTP_PASS)
            server.send_message(msg)
    except Exception as e:
        print("Error sending email:", e)


def format_date_str(d):
    try:
        return datetime.strptime(d, "%Y-%m-%d").strftime("%b %d, %Y")
    except Exception:
        return d

@app.route("/new", methods=["GET", "POST"])

        customer_name = request.form["customer_name"].strip()
        customer_email = request.form["customer_email"].strip()

        auth_notes = request.form.get("auth_notes", "").strip()
        stripe_customer_id = request.form.get("stripe_customer_id", "").strip()

        if not (customer_name and customer_email and sample_name and due_date):
            flash("Please fill in customer name, email, sample name, and due date.")
            return redirect(url_for("new_sample"))

        today = date.today().strftime("%Y-%m-%d")
        now_str = datetime.utcnow().isoformat()

        conn = get_db()
        c = conn.cursor()
        c.execute(
            """
            INSERT INTO samples (
                customer_name, customer_email, customer_phone,
                sample_name, checkout_date, due_date, status,
                notification_sent, team_notification_sent,

                created_at, updated_at

                customer_name,
                customer_email,
                customer_phone,
                sample_name,
                today,
                due_date,
                notes,
                auth_notes,
                stripe_customer_id,
                now_str,
                now_str,

        )
        conn.commit()
        conn.close()

                f"A new sample has been checked out.\n\n"

                f"Phone: {customer_phone}\n"

                f"Artisan Tile Sample Tracker"

        flash("Sample checkout saved.")
        return redirect(url_for("index"))

@app.route("/mark-returned/<int:sample_id>")
def mark_returned(sample_id):
    conn = get_db()
    c = conn.cursor()
    now_str = datetime.utcnow().isoformat()
    c.execute(
        "UPDATE samples SET status = 'returned', updated_at = ? WHERE id = ?",
        (now_str, sample_id),
    )
    conn.commit()
    conn.close()
    flash("Sample marked as returned.")
    return redirect(url_for("index"))


@app.route("/run-reminders")
def run_reminders():
    today_str = date.today().strftime("%Y-%m-%d")

    conn = get_db()
    c = conn.cursor()
    # Find checked-out samples that are past due and not yet notified
    c.execute(
        """
        SELECT * FROM samples
        WHERE status = 'checked_out'
          AND due_date < ?
          AND notification_sent = 0
        """,
        (today_str,),
    )
    overdue_samples = c.fetchall()

    count = 0
    for s in overdue_samples:
        customer_email = s["customer_email"]
        customer_name = s["customer_name"]
        sample_name = s["sample_name"]
        due_date = s["due_date"]

        subject = "Reminder to return your Artisan Tile sample"
        body = (
            f"Hi {customer_name},\n\n"
            f"This is a friendly reminder to return your sample from Artisan Tile.\n\n"
            f"Sample: {sample_name}\n"
            f"Due date: {format_date_str(due_date)}\n\n"
            f"If you have already returned it, you can ignore this message.\n\n"
            f"Thank you,\n"
            f"Artisan Tile"
        )
        send_email(customer_email, subject, body)

        # Notify team as well
        if TEAM_EMAIL and not s["team_notification_sent"]:
            t_subject = f"Overdue sample: {sample_name} for {customer_name}"
            t_body = (
                f"{TEAM_NAME},\n\n"
                f"The following sample is overdue:\n\n"
                f"Customer: {customer_name}\n"
                f"Email: {customer_email}\n"
                f"Phone: {s['customer_phone']}\n"
                f"Sample: {sample_name}\n"
                f"Due date: {format_date_str(due_date)}\n\n"
                f"Please follow up as needed.\n\n"
                f"Artisan Tile Sample Tracker"
            )
            send_email(TEAM_EMAIL, t_subject, t_body)
            c.execute(
                "UPDATE samples SET team_notification_sent = 1 WHERE id = ?",
                (s["id"],),
            )

        c.execute(
            "UPDATE samples SET notification_sent = 1, status = 'overdue', updated_at = ? WHERE id = ?",
            (datetime.utcnow().isoformat(), s["id"]),
        )
        count += 1

    conn.commit()
    conn.close()

    flash(f"Reminder check complete. {count} customer reminder(s) sent.")
    return redirect(url_for("index"))
